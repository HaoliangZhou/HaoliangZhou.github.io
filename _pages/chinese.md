  <section class="homepage-section research-summary-section" id="-research" aria-labelledby="research-title">
    <div class="research-summary">
      <h2 id="research-title">研究方向</h2>
      <p class="research-summary__intro">我目前的研究方向为多模态学习, 计算机视觉, 具体包括:</p>
      <div class="research-direction-list">
        <ul>
          <li><strong class="accent-text">组合式图像检索：</strong>结合视觉参考图像与自然语言修改的多模态指令中，准确检索符合用户意图的目标图像</li>
          <li><strong class="accent-text">多模态大语言模型：</strong>探索多模态大语言模型的感知、推理与交互能力，同时研究面向垂直领域的监督/指令微调以及基于强化学习的后训练方法</li>
          <li><strong class="accent-text">测试时自适应：</strong>通过持续学习和测试时自适应等技术，改进模型在部署后面对分布偏移、环境变化和未知数据时的鲁棒性与自适应能力</li>
          <li><strong class="accent-text">情感计算：</strong>建模并识别人脸表情、微表情及相关情感线索，提升模型对细粒度情绪变化的感知与理解能力</li>
        </ul>
      </div>
    </div>

document.addEventListener('DOMContentLoaded', function() {
  const page = document.querySelector('.home-page');
  if (!page) return;
  const themeToggle = document.getElementById('theme-toggle');
  const themeIcon = themeToggle ? themeToggle.querySelector('i') : null;
  const themeLabel = themeToggle ? themeToggle.querySelector('span') : null;
  const themeKey = 'haoliang-home-theme';
  const initialTheme = readStoredTheme() || 'light';

  applyTheme(initialTheme);

  if (themeToggle) {
    themeToggle.addEventListener('click', function() {
      const nextTheme = document.body.classList.contains('home-theme-dark') ? 'light' : 'dark';
      applyTheme(nextTheme);
      try {
        window.localStorage.setItem(themeKey, nextTheme);
      } catch (error) {
        // Local storage can be unavailable in private browsing contexts.
      }
    });
  }

  function readStoredTheme() {
    try {
      const theme = window.localStorage.getItem(themeKey);
      return theme === 'dark' || theme === 'light' ? theme : null;
    } catch (error) {
      return null;
    }
  }

  function applyTheme(theme) {
    const isDark = theme === 'dark';
    document.body.classList.toggle('home-theme-dark', isDark);
    document.body.classList.toggle('home-theme-light', !isDark);
    document.documentElement.style.colorScheme = isDark ? 'dark' : 'light';

    if (!themeToggle) return;
    themeToggle.setAttribute('aria-pressed', String(isDark));
    themeToggle.setAttribute('aria-label', isDark ? '切换到白天模式' : '切换到夜间模式');
    if (themeIcon) {
      themeIcon.classList.toggle('fa-moon', !isDark);
      themeIcon.classList.toggle('fa-sun', isDark);
    }
    if (themeLabel) {
      themeLabel.textContent = isDark ? '白天' : '夜间';
    }
  }

  limitScrollableSections(6);
  window.addEventListener('load', function() {
    limitScrollableSections(6);
  });
  window.addEventListener('resize', function() {
    limitScrollableSections(6);
  }, { passive: true });

  function limitScrollableSections(visibleCount) {
    limitScrollableList('.news-timeline', '.news-item', visibleCount, 'news-timeline--scrollable');
    limitScrollableList('.award-rail', '.achievement-card', visibleCount, 'award-rail--scrollable');
    toggleOverflowHint('.service-project-rail');
  }

  function limitScrollableList(listSelector, itemSelector, visibleCount, scrollableClass) {
    document.querySelectorAll(listSelector).forEach(function(list) {
      const items = Array.from(list.children).filter(function(child) {
        return child.matches(itemSelector);
      });
      const hint = list.nextElementSibling && list.nextElementSibling.classList.contains('scroll-more-hint')
        ? list.nextElementSibling
        : null;

      list.style.maxHeight = '';
      list.classList.remove(scrollableClass);
      list.scrollTop = 0;
      if (hint) hint.hidden = true;

      if (items.length <= visibleCount) return;

      const nextItem = items[visibleCount];
      const maxHeight = nextItem.getBoundingClientRect().top - list.getBoundingClientRect().top;
      list.style.maxHeight = Math.max(0, maxHeight).toFixed(2) + 'px';
      list.classList.add(scrollableClass);
      if (hint) hint.hidden = false;
    });
  }

  function toggleOverflowHint(listSelector) {
    document.querySelectorAll(listSelector).forEach(function(list) {
      const hint = list.nextElementSibling && list.nextElementSibling.classList.contains('scroll-more-hint')
        ? list.nextElementSibling
        : null;
      if (!hint) return;

      hint.hidden = list.scrollWidth <= list.clientWidth + 1;
    });
  }

  const animatedItems = page.querySelectorAll('.news-item, .experience-item, .research-card, .achievement-card, .info-panel');
  if ('IntersectionObserver' in window) {
    const observer = new IntersectionObserver(function(entries) {
      entries.forEach(function(entry) {
        if (entry.isIntersecting) {
          entry.target.classList.add('is-visible');
          observer.unobserve(entry.target);
        }
      });
    }, { threshold: 0.12, rootMargin: '0px 0px -60px 0px' });
    animatedItems.forEach(function(item) {
      item.classList.add('reveal-on-scroll');
      observer.observe(item);
    });
  } else {
    animatedItems.forEach(function(item) {
      item.classList.add('is-visible');
    });
  }

  const wrapper = document.getElementById('publications-wrapper');
  if (wrapper) {
    const filterContainer = document.getElementById('filter-container');
    const searchInput = document.getElementById('publication-search-input');
    const resetButton = document.getElementById('publication-reset');
    const countLabel = document.getElementById('publication-count');
    const paperBoxes = Array.from(wrapper.querySelectorAll('.paper-box'));
    const activeTags = new Set();
    const tagCounts = {};
    const visibleTagsWhitelist = [
      'CCF-A',
      '中科院一区',
      '第一作者',
      'Composed Image Retrieval',
      'Continual Learning',
      'Facial Expression Recognition',
      'Micro Expression Recognition'
    ];

    paperBoxes.forEach(function(box) {
      const tags = getTags(box);
      const textContainer = box.querySelector('.paper-box-text');
      const linksContainer = box.querySelector('.links');

      if (textContainer && !textContainer.querySelector('.badge-container')) {
        const badgeContainer = document.createElement('div');
        badgeContainer.className = 'badge-container';
        tags.forEach(function(tag) {
          const badge = document.createElement('span');
          badge.className = 'inner-tag-badge';
          badge.textContent = tag;
          badgeContainer.appendChild(badge);
        });
        if (linksContainer) {
          textContainer.insertBefore(badgeContainer, linksContainer);
        } else {
          textContainer.appendChild(badgeContainer);
        }
      }

      tags.forEach(function(tag) {
        if (visibleTagsWhitelist.includes(tag)) {
          tagCounts[tag] = (tagCounts[tag] || 0) + 1;
        }
      });
    });

    visibleTagsWhitelist
      .filter(function(tag) { return tagCounts[tag]; })
      .forEach(function(tag) {
        const button = document.createElement('button');
        button.className = 'filter-btn';
        button.type = 'button';
        button.textContent = tag + ' (' + tagCounts[tag] + ')';
        button.addEventListener('click', function() {
          if (activeTags.has(tag)) {
            activeTags.delete(tag);
            button.classList.remove('active');
          } else {
            activeTags.add(tag);
            button.classList.add('active');
          }
          renderPublications();
        });
        filterContainer.appendChild(button);
      });

    if (searchInput) {
      searchInput.addEventListener('input', renderPublications);
    }

    if (resetButton) {
      resetButton.addEventListener('click', function() {
        activeTags.clear();
        filterContainer.querySelectorAll('.filter-btn').forEach(function(button) {
          button.classList.remove('active');
        });
        if (searchInput) searchInput.value = '';
        renderPublications();
      });
    }

    renderPublications();

    function getTags(box) {
      const rawTags = box.getAttribute('data-tags') || '';
      return rawTags.split(',').map(function(tag) { return tag.trim(); }).filter(Boolean);
    }

    function renderPublications() {
      const query = searchInput ? searchInput.value.trim().toLowerCase() : '';
      let visibleCount = 0;

      paperBoxes.forEach(function(box) {
        const tags = getTags(box);
        const text = box.textContent.toLowerCase();
        const matchesTags = activeTags.size === 0 || Array.from(activeTags).every(function(tag) {
          return tags.includes(tag);
        });
        const matchesQuery = !query || text.includes(query);
        const shouldShow = matchesTags && matchesQuery;

        box.classList.toggle('hidden', !shouldShow);
        box.querySelectorAll('.inner-tag-badge').forEach(function(badge) {
          badge.classList.toggle('active', activeTags.has(badge.textContent));
        });
        if (shouldShow) visibleCount += 1;
      });

      if (countLabel) {
        countLabel.textContent = visibleCount + ' / ' + paperBoxes.length + ' 篇论文显示';
      }
    }
  }

  const backToTop = document.getElementById('back-to-top');
  if (backToTop) {
    window.addEventListener('scroll', function() {
      backToTop.classList.toggle('is-visible', window.scrollY > 700);
    }, { passive: true });
    backToTop.addEventListener('click', function() {
      window.scrollTo({ top: 0, behavior: 'smooth' });
    });
  }
});
</script>
